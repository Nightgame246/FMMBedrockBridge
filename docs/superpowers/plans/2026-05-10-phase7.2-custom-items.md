# Phase 7.2 Custom Items — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** EliteMobs Custom Items (Waffen, Rüstung, GUI-Icons) auf Bedrock-Clients sichtbar machen, indem wir `custom_model_data`-Items über Geysers `GeyserDefineCustomItemsEvent` registrieren und die PNG-Texturen in unser Bedrock-Pack einbetten.

**Architecture:** Der Spigot-Plugin-seitige `EliteMobsItemScanner` liest das EliteMobs Java Resource Pack bei Server-Start, extrahiert alle `custom_model_data`-Mappings und schreibt sie als `em-items.json` + Textur-Dateien in den `bedrock-pack/`-Ordner. Die Geyser-Extension liest diese Dateien beim Start und (a) bettet die Texturen in das generierte Bedrock-Pack ein und (b) registriert jeden Eintrag beim `GeyserDefineCustomItemsEvent`. Phase 7.2a ist reine SSH-Recherche ohne Code — die Ergebnisse bestätigen die in 7.2b verwendeten Pfade und Felder.

**Tech Stack:** Java 21, Paper API 1.21.x, Geyser API 2.7.0-SNAPSHOT (`GeyserDefineCustomItemsEvent`, `CustomItemData`, `CustomItemOptions`), Gson, Maven

- [ ] **Scanner-Klasse anlegen**

```java
package de.crazypandas.fmmbedrockbridge.bridge;

import com.google.gson.Gson;
import com.google.gson.JsonArray;
import com.google.gson.JsonElement;
import com.google.gson.JsonObject;
import de.crazypandas.fmmbedrockbridge.FMMBedrockBridge;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.List;
import java.util.logging.Logger;

/**
 * Scans the EliteMobs Java resource pack for custom_model_data overrides
 * and maps them to EMCustomItem records.
 *
 * Expected pack structure:
 *   assets/minecraft/models/item/<base>.json  — has "overrides" with custom_model_data predicates
 *   assets/<ns>/models/item/<name>.json       — override target, has "textures.layer0"
 *   assets/<ns>/textures/item/<name>.png      — the actual texture
 */
public class EliteMobsItemScanner {

    private static final Logger log = FMMBedrockBridge.getInstance().getLogger();
    private static final Gson GSON = new Gson();

    private final Path packRoot;

    public EliteMobsItemScanner(Path packRoot) {
        this.packRoot = packRoot;
    }

    /**
     * Scans the pack and returns all found custom item mappings.
     * Returns empty list if pack root doesn't exist or no overrides found.
     */
    public List<EMCustomItem> scan() {
        List<EMCustomItem> result = new ArrayList<>();
        Path itemModelsDir = packRoot.resolve("assets/minecraft/models/item");

        if (!Files.isDirectory(itemModelsDir)) {
            log.warning("[ItemScanner] EM resource pack not found at " + packRoot + " — skipping custom item scan.");
            return result;
        }

        try (var stream = Files.list(itemModelsDir)) {
            stream.filter(p -> p.toString().endsWith(".json"))
                    .forEach(modelFile -> scanModelFile(modelFile, result));
        } catch (IOException e) {
            log.warning("[ItemScanner] Failed to list item models: " + e.getMessage());
        }

        log.info("[ItemScanner] Found " + result.size() + " EM custom items.");
        return result;
    }

    private void scanModelFile(Path modelFile, List<EMCustomItem> result) {
        try {
            String baseMaterial = "minecraft:" + modelFile.getFileName().toString().replace(".json", "");
            String json = Files.readString(modelFile);
            JsonObject root = GSON.fromJson(json, JsonObject.class);

            if (!root.has("overrides")) return;
            JsonArray overrides = root.getAsJsonArray("overrides");

            for (JsonElement el : overrides) {
                JsonObject override = el.getAsJsonObject();
                JsonObject predicate = override.getAsJsonObject("predicate");
                if (predicate == null || !predicate.has("custom_model_data")) continue;

                int cmd = predicate.get("custom_model_data").getAsInt();
                String modelRef = override.get("model").getAsString(); // e.g. "elitemobs:item/em_sword_1"

                Path texturePath = resolveTexture(modelRef);
                if (texturePath == null) continue;

                String textureKey = "em/" + baseMaterial.replace("minecraft:", "") + "_" + cmd;
                result.add(new EMCustomItem(baseMaterial, cmd, texturePath.toString(), textureKey));
            }
        } catch (Exception e) {
            log.warning("[ItemScanner] Error scanning " + modelFile + ": " + e.getMessage());
        }
    }

    /**
     * Follows a model reference like "elitemobs:item/em_sword_1" to find the PNG.
     * Looks for the texture in assets/<namespace>/textures/item/<name>.png
     */
    private Path resolveTexture(String modelRef) {
        // modelRef: "namespace:item/name"
        String[] parts = modelRef.split(":");
        if (parts.length != 2) return null;
        String namespace = parts[0];
        String itemPath = parts[1]; // e.g. "item/em_sword_1"

        // Read the override target model to find textures.layer0
        Path overrideModelFile = packRoot.resolve("assets/" + namespace + "/models/" + itemPath + ".json");
        if (!Files.exists(overrideModelFile)) return null;

        try {
            String json = Files.readString(overrideModelFile);
            JsonObject model = GSON.fromJson(json, JsonObject.class);
            if (!model.has("textures")) return null;

            JsonObject textures = model.getAsJsonObject("textures");
            String layer0 = null;
            if (textures.has("layer0")) layer0 = textures.get("layer0").getAsString();
            else if (textures.has("all")) layer0 = textures.get("all").getAsString();
            if (layer0 == null) return null;

            // layer0: "namespace:item/name" → assets/namespace/textures/item/name.png
            String[] texParts = layer0.split(":");
            String texNs = texParts.length == 2 ? texParts[0] : namespace;
            String texPath = texParts.length == 2 ? texParts[1] : texParts[0];

            Path pngPath = packRoot.resolve("assets/" + texNs + "/textures/" + texPath + ".png");
            if (!Files.exists(pngPath)) return null;
            return pngPath;
        } catch (Exception e) {
            return null;
        }
    }
}
```

    /**
     * Reads em-items.json from the input dir, copies textures into the pack,
     * and appends entries to textures/item_texture.json.
     * Call this after all entity model files have been processed, before zip().
     */
    public List<EmItemEntry> embedEliteItems(Path inputDir) {
        Path emItemsJson = inputDir.resolve("em-items.json");
        if (!Files.exists(emItemsJson)) {
            extension.logger().info("[Phase 7.2b] No em-items.json found — skipping custom item embedding.");
            return List.of();
        }

        List<EmItemEntry> entries = new ArrayList<>();
        try {
            String json = Files.readString(emItemsJson);
            com.google.gson.reflect.TypeToken<List<EmItemEntry>> token = new com.google.gson.reflect.TypeToken<>() {};
            entries = GSON.fromJson(json, token.getType());

            Path itemTexturesDir = packDir.resolve("textures/items");
            Files.createDirectories(itemTexturesDir);

            Map<String, Object> textureData = new java.util.LinkedHashMap<>();

            for (EmItemEntry entry : entries) {
                // Copy PNG
                Path srcTexture = inputDir.resolve(entry.sourceTexturePath());
                if (!Files.exists(srcTexture)) {
                    extension.logger().warning("[Phase 7.2b] Texture not found: " + srcTexture);
                    continue;
                }
                // Flatten texture key: "em/diamond_sword_1001" → "em_diamond_sword_1001.png"
                String bedrockFileName = entry.bedrockTextureKey().replace("/", "_");
                Path dest = itemTexturesDir.resolve(bedrockFileName + ".png");
                Files.copy(srcTexture, dest, java.nio.file.StandardCopyOption.REPLACE_EXISTING);

                // Add to item_texture map: key → textures/items/<filename (no extension)>
                Map<String, String> textureEntry = Map.of("textures", "textures/items/" + bedrockFileName);
                textureData.put(entry.bedrockTextureKey(), textureEntry);
            }

            // Write / merge item_texture.json
            Path itemTexturePath = packDir.resolve("textures/item_texture.json");
            Map<String, Object> itemTextureRoot = new java.util.LinkedHashMap<>();
            itemTextureRoot.put("resource_pack_name", "fmmbridge");
            itemTextureRoot.put("texture_data", textureData);
            writeJson(itemTexturePath, itemTextureRoot);

            extension.logger().info("[Phase 7.2b] Embedded " + entries.size() + " EM item textures into Bedrock pack.");
        } catch (Exception e) {
            extension.logger().error("[Phase 7.2b] Failed to embed elite items: " + e.getMessage(), e);
        }
        return entries;
    }

    /** Mirrors EMCustomItem record for deserialization inside the extension. */
    public record EmItemEntry(
            String javaMaterial,
            int customModelData,
            String sourceTexturePath,
            String bedrockTextureKey
    ) {}
```

    @Subscribe
    public void onDefineCustomItems(org.geysermc.geyser.api.event.lifecycle.GeyserDefineCustomItemsEvent event) {
        Path emItemsJson = this.dataFolder().resolve("input/em-items.json");
        if (!Files.exists(emItemsJson)) {
            this.logger().info("[Phase 7.2b] No em-items.json in input/ — no EM custom items registered.");
            return;
        }

        try {
            String json = Files.readString(emItemsJson);
            com.google.gson.reflect.TypeToken<java.util.List<ResourcePackBuilder.EmItemEntry>> token =
                    new com.google.gson.reflect.TypeToken<>() {};
            java.util.List<ResourcePackBuilder.EmItemEntry> entries = GSON.fromJson(json, token.getType());

            int registered = 0;
            for (ResourcePackBuilder.EmItemEntry entry : entries) {
                try {
                    org.geysermc.geyser.api.item.custom.CustomItemData itemData =
                            org.geysermc.geyser.api.item.custom.CustomItemData.builder()
                                    .name(entry.bedrockTextureKey().replace("/", "_"))
                                    .customItemOptions(
                                            org.geysermc.geyser.api.item.custom.CustomItemOptions.builder()
                                                    .customModelData(entry.customModelData())
                                                    .build())
                                    .icon(entry.bedrockTextureKey())
                                    .build();

                    event.register(entry.javaMaterial(), itemData);
                    registered++;
                } catch (Exception e) {
                    this.logger().warning("[Phase 7.2b] Failed to register item " + entry.bedrockTextureKey() + ": " + e.getMessage());
                }
            }
            this.logger().info("[Phase 7.2b] Registered " + registered + " EM custom items with Geyser.");
        } catch (Exception e) {
            this.logger().error("[Phase 7.2b] Failed to read em-items.json: " + e.getMessage(), e);
        }
    }
```
